1)mkdir test_dir=creates a directory named test_dir
2)cd test_dir=enters in the directory test_dir
3)touch example.txt=creates a file name example.txt in test_dir directory
4)mv example.txt renamed_example.txt=renames the file
5)cat=displays the content of the file
6)head -n 5/etc/passwd=give first 5 lines of the content
7)tail -n 5 /etc/passwd=gives last 5 lines oof the content of the file
8)grep "root" /etc/passwd=gives all the lines containing the word "root"
9)zip -r test_dir.zip test_dir=creates a new zipped directory called test_dir.zip
10)unzip test_dir.zip -d unzipped_dir=creates a new unzipped directory having the contents of zipped directory test_dir.zip
11)wget=used to download  content from links
12)chmod 444 secure.txt=changes permission to read-only for everyone
13)export MY_VAR="ab"    =changes the value of MY_VAR to ab
14)echo $MY_VAR=displays the value of MY_VAR
